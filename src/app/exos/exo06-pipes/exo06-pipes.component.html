Exercice 06 - Pipes

String Pipe

{{ myString }} => UpperCase
{{ myString | uppercase }}

{{ myString }} => lowercase
{{ myString | lowercase }}

{{ myString }} => Title Case
{{ myString | titlecase }}

Number Pipe
number : minIntegerDigits . minFractionDigits - maxFractionDigits

{{ myNumber1 }} => number : '1.2-3'
{{ myNumber1 | number : '1.2-3' }}

{{ myNumber2 }} => number : '1.2-3'
{{ myNumber2 | number : '1.2-3' }}

Percent Pipe

{{ myPercentage }} => myPercentage : '1.2-3'
{{ myPercentage | percent : '1.2-3' }}

Currency Pipe

{{ myNumber1 }} => currency : 'EUR' : 'symbol' : '1.2-2'
{{ myNumber1 | currency : 'EUR' : 'symbol' : '1.2-2' }}
{{ myNumber1 | currency : 'EUR' : 'code' : '1.2-2' }}

Date Pipe

2020/1/1 => myDate: 'EEEE/dd/MM/yyyy'
{{ myDate | date : 'EEEE/dd/MM/yyyy' }}

2020/1/1 => myDate: 'dd/MMMM/yyyy'
{{ myDate | date : 'dd/MMMM/yyyy' }}

JSON Pipe

{{ myObject }} => JSON
{{ myObject | json }}
